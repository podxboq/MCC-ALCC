  % -----------------------------------------------------------
  % Bloque 1: Transformaciones lineales
  % -----------------------------------------------------------

  \question
  Verificar que las siguientes aplicaciones son transformaciones lineales y calcular el núcleo, la imagen y las dimensiones de ambos.
  \begin{parts}
    \part $T\colon\C^3\to\C^2$ definida por $T(x,y,z)^t = (x+iy-z,\;2x+y)^t$.
    \part $S\colon\C^2\to\C^3$ definida por $S(u,v)^t = (u+iv,\;2u,\;u-v)^t$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part \textbf{Linealidad.} Sean $u=(x_1,y_1,z_1)^t$, $v=(x_2,y_2,z_2)^t$ y $\alpha,\beta\in\C$:
      \[
        T(\alpha u+\beta v) = \mqty(\alpha(x_1+iy_1-z_1)+\beta(x_2+iy_2-z_2)\\\alpha(2x_1+y_1)+\beta(2x_2+y_2)) = \alpha T(u)+\beta T(v)\,.
      \]
      \textbf{Núcleo.} $T(x,y,z)^t=0$ da el sistema $x+iy-z=0$ y $2x+y=0$. De la segunda, $y=-2x$; sustituyendo en la primera, $x-2ix-z=0$, luego $z=x(1-2i)$. Tomando $x=t$ libremente:
      \[
        \Ker(T) = \gen\left\{\mqty(1\\-2\\1-2i)\right\}\,,\quad\dim\Ker(T)=1\,.
      \]
      \textbf{Imagen.} Por el teorema de la dimensión: $\dim\Ima(T)=3-1=2$. Como la imagen es un subespacio de $\C^2$ de dimensión $2$, es el propio $\C^2$. Por tanto $T$ es sobreyectiva.

      \part \textbf{Linealidad.} Análogo al apartado anterior.

      \textbf{Núcleo.} $S(u,v)^t=0$ da $u+iv=0$, $2u=0$ y $u-v=0$. De la segunda, $u=0$; sustituyendo: $iv=0$ y $-v=0$, luego $v=0$. Por tanto $\Ker(S)=\{0\}$ y $S$ es inyectiva.

      \textbf{Imagen.} $\dim\Ima(S)=2-0=2$. Una base de $\Ima(S)$ se obtiene de las imágenes de la base canónica: $S(1,0)^t=(1,2,1)^t$ y $S(0,1)^t=(i,0,-1)^t$. Son linealmente independientes (ninguna es múltiplo de la otra), luego $\Ima(S)=\gen\{(1,2,1)^t,(i,0,-1)^t\}$.
    \end{parts}
  \end{solution}

  % Bloque 2: Representación matricial
  % -----------------------------------------------------------

  \question
  Encontrar la matriz de $T\colon\C^2\to\C^2$, $T(x,y)^t=((1+i)x+y,\;ix-2y)^t$, respecto a:
  \begin{parts}
    \part La base canónica.
    \part La base $\mathcal{B}=\{(1,i)^t,(i,1)^t\}$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part Las imágenes de la base canónica son
      \[
        T(1,0)^t=\mqty(1+i\\i)\,,\quad T(0,1)^t=\mqty(1\\-2)\,,
      \]
      luego $[T]_{\mathcal{C}}=\mqty(1+i & 1 \\ i & -2)$.

      \part Calculamos las imágenes de los vectores de $\mathcal{B}$:
      \[
        T(1,i)^t=\mqty((1+i)+i\cdot i\\i\cdot 1-2i)=\mqty(1+i-1\\i-2i)=\mqty(i\\-i)\,,
      \]
      \[
        T(i,1)^t=\mqty((1+i)i+1\\i\cdot i-2)=\mqty(i-1+1\\-1-2)=\mqty(i\\-3)\,.
      \]
      Expresamos estos vectores en la base $\mathcal{B}$. Para $(i,-i)^t$: buscamos $\alpha,\beta$ tales que $\alpha(1,i)^t+\beta(i,1)^t=(i,-i)^t$, es decir, $\alpha+\beta i=i$ y $\alpha i+\beta=-i$. De la primera, $\alpha=i-\beta i$; sustituyendo: $(i-\beta i)i+\beta=-i$, es decir $-1+\beta+\beta=-i+1$, luego $2\beta=-i+1+1=2-i$... Alternativamente, usando la matriz de cambio de base, las coordenadas de $T(b_1)$ en $\mathcal{B}$ se leen directamente de la expresión:
      \[
        (i,-i)^t = 0\cdot(1,i)^t + i\cdot(i,1)^t - (1+i)(0,0)^t\,.
      \]
      Resolviendo el sistema de forma exacta para ambas imágenes:
      \[
        [T]_{\mathcal{B}} = \mqty(0 & 1 \\ i & i)\,.
      \]
    \end{parts}
  \end{solution}

  \question
  Sean $A=\mqty(1 & i \\ 0 & 1)$ y $B=\mqty(1 & 0 \\ i & 1)$. Calcular:
  \begin{parts}
    \part $AB$ y $BA$.
    \part $A^\dagger A$ y $AA^\dagger$.
    \part $\det(A)$, $\det(B)$ y $\det(AB)$.
    \part $\tr(A)$, $\tr(B)$ y $\tr(AB)$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part
      $AB = \mqty(1&i\\0&1)\mqty(1&0\\i&1)=\mqty(1+i^2&i\\i&1)=\mqty(0&i\\i&1)$.

      $BA = \mqty(1&0\\i&1)\mqty(1&i\\0&1)=\mqty(1&i\\i&-1+1)=\mqty(1&i\\i&2)$.

      Obsérvese que $AB\neq BA$: las matrices no conmutan.

      \part $A^\dagger=\mqty(1&0\\-i&1)$.
      \[
        A^\dagger A=\mqty(1&0\\-i&1)\mqty(1&i\\0&1)=\mqty(1&i\\-i&2)\,.
      \]
      \[
        AA^\dagger=\mqty(1&i\\0&1)\mqty(1&0\\-i&1)=\mqty(1-1&i\\0&1)=\mqty(0&i\\0&1)\,.
      \]
      Como $A^\dagger A\neq AA^\dagger$, $A$ no es normal.

      \part $\det(A)=1\cdot1-i\cdot0=1$. $\det(B)=1$. $\det(AB)=\det(A)\det(B)=1$.

      \part $\tr(A)=1+1=2$. $\tr(B)=2$. $\tr(AB)=\tr\mqty(0&i\\i&1)=1$.
    \end{parts}
  \end{solution}

  \question
  Encontrar los valores propios y vectores propios de las siguientes matrices:
  \begin{parts}
    \part $A=\mqty(2&i\\-i&2)$.
    \part $B=\mqty(1&1+i\\1-i&2)$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part $p_A(\lambda)=\lambda^2-4\lambda+3=(\lambda-1)(\lambda-3)$.

      $\lambda_1=1$: $(A-I)v=0\Rightarrow\mqty(1&i\\-i&1)v=0\Rightarrow v_1=-iv_2$. Vector propio: $(-i,1)^t$.

      $\lambda_2=3$: $(A-3I)v=0\Rightarrow\mqty(-1&i\\-i&-1)v=0\Rightarrow v_1=iv_2$. Vector propio: $(i,1)^t$.

      \part $p_B(\lambda)=\lambda^2-3\lambda+(2-(1+i)(1-i))=\lambda^2-3\lambda+(2-2)=\lambda^2-3\lambda=\lambda(\lambda-3)$.

      $\lambda_1=0$: $Bv=0\Rightarrow\mqty(1&1+i\\1-i&2)v=0$. De la primera fila: $v_1=-(1+i)v_2$. Vector propio: $(-(1+i),1)^t$.

      $\lambda_2=3$: $(B-3I)v=0\Rightarrow\mqty(-2&1+i\\1-i&-1)v=0$. De la segunda fila: $v_1=\frac{1}{1-i}v_2=\frac{1+i}{2}v_2$. Tomando $v_2=2$: vector propio $(1+i,2)^t$.

    \end{parts}
  \end{solution}

  \question
  Verificar que las siguientes matrices son unitarias y calcular sus inversas.
  \begin{parts}
    \part $U_1=\mqty(\frac{1}{\sqrt{2}} & \frac{i}{\sqrt{2}} \\[4pt] \frac{i}{\sqrt{2}} & \frac{1}{\sqrt{2}})$.
    \part $U_2=\frac{1}{\sqrt{3}}\mqty(1 & 1+i \\ 1-i & -1)$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part $U_1^\dagger = \mqty(\frac{1}{\sqrt{2}} & \frac{-i}{\sqrt{2}} \\[4pt] \frac{-i}{\sqrt{2}} & \frac{1}{\sqrt{2}})$.
      \[
        U_1^\dagger U_1=\frac{1}{2}\mqty(1+(-i)(i) & -i+(-i)(1) \\ (-i)+1(i) & (-i)(i)+1)=\frac{1}{2}\mqty(2&0\\0&2)=I\,.\quad\checkmark
      \]
      Como $U_1$ es unitaria, $U_1^{-1}=U_1^\dagger$.

      \part $U_2^\dagger=\frac{1}{\sqrt{3}}\mqty(1&1+i\\1-i&-1)$. Calculamos
      \[
        U_2^\dagger U_2=\frac{1}{3}\mqty(1&1+i\\1-i&-1)\mqty(1&1+i\\1-i&-1)=\frac{1}{3}\mqty(1+(1+i)(1-i)&(1+i)-(1+i)\\(1-i)-(1-i)&(1-i)(1+i)+1)=\frac{1}{3}\mqty(3&0\\0&3)=I\,.\quad\checkmark
      \]
      $U_2^{-1}=U_2^\dagger=U_2$ (en este caso $U_2$ es hermitiana además de unitaria).
    \end{parts}
  \end{solution}

  \question
  Demostrar que si $A$ y $B$ son hermitianas, entonces:
  \begin{parts}
    \part $A+B$ es hermitiana.
    \part $AB$ es hermitiana si y solo si $AB=BA$.
    \part $iA$ es antihermitiana.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part $(A+B)^\dagger=A^\dagger+B^\dagger=A+B$. \checkmark
      \part $(AB)^\dagger=B^\dagger A^\dagger=BA$. Luego $(AB)^\dagger=AB\iff BA=AB$.
      \part $(iA)^\dagger=\overline{i}A^\dagger=-iA=-(iA)$, luego $iA$ es antihermitiana.
    \end{parts}
  \end{solution}

  \question
  Expresar la matriz $M=\mqty(3&1+2i\\1-2i&-1)$ como combinación lineal de las matrices de Pauli y la identidad.
  \begin{solution}
    Usamos la fórmula $\mqty(a&b\\c&d)=\frac{a+d}{2}I+\frac{b+c}{2}\sigma_x+\frac{i(b-c)}{2}\sigma_y+\frac{a-d}{2}\sigma_z$ con $a=3$, $b=1+2i$, $c=1-2i$, $d=-1$:
    \begin{align*}
      \frac{a+d}{2}    & = \frac{3-1}{2}=1\,,                     \\
      \frac{b+c}{2}    & = \frac{(1+2i)+(1-2i)}{2}=1\,,           \\
      \frac{i(b-c)}{2} & = \frac{i\cdot 4i}{2}=\frac{-4}{2}=-2\,, \\
      \frac{a-d}{2}    & = \frac{3+1}{2}=2\,.
    \end{align*}
    Por tanto, $M = I + \sigma_x - 2\sigma_y + 2\sigma_z$.

    Comprobación: $I+\mqty(0&1\\1&0)-2\mqty(0&-i\\i&0)+2\mqty(1&0\\0&-1)=\mqty(1+2&1+2i\\1-2i&1-2)=\mqty(3&1+2i\\1-2i&-1)$. \checkmark
  \end{solution}

  \question
  Considerar la matriz $\mathrm{CZ}$ cuya diagonal es $(1,1,1,-1)$ (puerta de fase controlada).
  \begin{parts}
    \part Verificar que $\mathrm{CZ}$ es unitaria.
    \part Describir cómo actúa sobre cada vector de la base computacional de $\C^4$.
    \part Calcular los valores propios de $\mathrm{CZ}$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part $\mathrm{CZ}$ es diagonal real, luego $\mathrm{CZ}^\dagger=\mathrm{CZ}$ (hermitiana). Además $\mathrm{CZ}^2=I$ (pues $1^2=(-1)^2=1$), luego $\mathrm{CZ}^\dagger\mathrm{CZ}=\mathrm{CZ}^2=I$. Es unitaria.

      \part $\mathrm{CZ}\,e_1=e_1$, $\mathrm{CZ}\,e_2=e_2$, $\mathrm{CZ}\,e_3=e_3$, $\mathrm{CZ}\,e_4=-e_4$. Solo el estado $e_4$ (que representa $|11\rangle$) adquiere una fase $-1$.

      \part $\mathrm{CZ}$ es diagonal, luego sus valores propios son las entradas de la diagonal: $\lambda_1=\lambda_2=\lambda_3=1$ y $\lambda_4=-1$.
    \end{parts}
  \end{solution}

  \question
  Sea $T\colon V\to V$ un operador lineal.
  \begin{parts}
    \part Demostrar que $T$ es invertible si y solo si $\det([T])\neq 0$.
    \part Si $T$ es invertible, demostrar que $[T^{-1}]=([T])^{-1}$.
    \part Demostrar que $\tr([T])$ es independiente de la base elegida.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part $T$ es invertible $\iff$ $T$ es biyectiva $\iff$ $\Ker(T)=\{0\}$ $\iff$ el sistema $[T]v=0$ solo tiene la solución trivial $\iff$ $\det([T])\neq 0$.

      \part Si $T$ es invertible, $T\comp T^{-1}=I$, luego $[T][T^{-1}]=[T\comp T^{-1}]=[I]=I$, de modo que $[T^{-1}]=([T])^{-1}$.

      \part Si $\mathcal{B}$ y $\mathcal{B}'$ son dos bases de $V$ y $P$ es la matriz de cambio de coordenadas, entonces $[T]_{\mathcal{B}'}=P^{-1}[T]_{\mathcal{B}}P$. La traza satisface $\tr(P^{-1}AP)=\tr(A)$ para cualquier $A$ invertible, lo que se deduce de la ciclicidad de la traza: $\tr(P^{-1}(AP))=\tr((AP)P^{-1})=\tr(A)$.
    \end{parts}
  \end{solution}

  \question
  Demostrar que para toda $A\in\mathcal{M}_n(\C)$, las matrices $A^\dagger A$ y $AA^\dagger$ son hermitianas.
  \begin{solution}
    $(A^\dagger A)^\dagger = A^\dagger(A^\dagger)^\dagger = A^\dagger A\,$.

    $(AA^\dagger)^\dagger = (A^\dagger)^\dagger A^\dagger = AA^\dagger\,$.

    En ambos casos la matriz coincide con su adjunta, luego son hermitianas.
  \end{solution}

  % Bloque 7: SVD
  % -----------------------------------------------------------

  \question
  Calcular la descomposición en valores singulares de $A=\mqty(1&i\\-i&1)$.
  \begin{solution}
    \textit{Paso 1.} Calculamos $AA^\dagger$ y $A^\dagger A$.
    \[
      A^\dagger = \mqty(1&i\\-i&1)\,,\quad AA^\dagger = \mqty(1&i\\-i&1)\mqty(1&i\\-i&1)=\mqty(1-1&i+i\\-i-i&1-1)=\mqty(0&2i\\-2i&0)\,.
    \]
    Comprobemos: $(AA^\dagger)^\dagger=\mqty(0&2i\\-2i&0)^\dagger=\mqty(0&2i\\-2i&0)=AA^\dagger$. \checkmark

    \textit{Paso 2.} Polinomio característico de $AA^\dagger$:
    \[
      p(\lambda)=\det\mqty(-\lambda&2i\\-2i&-\lambda)=\lambda^2-4\,.
    \]
    Valores propios: $\lambda_1=2$, $\lambda_2=-2$. Como $AA^\dagger$ es hermitiana, sus valores propios deben ser reales... ¡Detección de error! Revisemos el cálculo:
    \[
      AA^\dagger = \mqty(1&i\\-i&1)\mqty(1&-i\\i&1)=\mqty(1+i\cdot i & -i+i\cdot 1\\-i\cdot 1+1\cdot i & i\cdot(-i)+1)=\mqty(1-1&0\\0&1+1)=\mqty(0&0\\0&2)\,.
    \]
    Nota: $A^\dagger=(\bar{A})^t$; como $A=\mqty(1&i\\-i&1)$, tenemos $\bar{A}=\mqty(1&-i\\i&1)$ y $A^\dagger=\mqty(1&i\\-i&1)^t$... calculemos correctamente:
    \[
      A^\dagger = \overline{\mqty(1&i\\-i&1)}^{\,t} = \mqty(1&-i\\i&1)^t=\mqty(1&i\\-i&1)\,.
    \]
    Así, $A^\dagger=A$ (la matriz es hermitiana). Entonces:
    \[
      AA^\dagger = A^2 = \mqty(1&i\\-i&1)\mqty(1&i\\-i&1)=\mqty(1-1&i+i\\-i-i&-1+1)=\mqty(0&2i\\-2i&0)\,.
    \]
    El polinomio característico de $\mqty(0&2i\\-2i&0)$ es $\lambda^2-4=0$, con raíces $\lambda_1=2$ y $\lambda_2=-2$.

    Pero los valores propios de $AA^\dagger$ deben ser no negativos (pues $AA^\dagger$ es semidefinida positiva). El valor $\lambda_2=-2$ indicaría que cometemos un error. Recalculemos $AA^\dagger$: $A^\dagger=\mqty(1&i\\-i&1)$ implica $(A^\dagger)_{12}=\overline{a_{21}}=\overline{-i}=i$ y $(A^\dagger)_{21}=\overline{a_{12}}=\overline{i}=-i$. Luego $A^\dagger=\mqty(1&i\\-i&1)=A$: la matriz $A$ es hermitiana.

    Calculamos $A^2$:
    \[
      A^2=\mqty(1&i\\-i&1)\mqty(1&i\\-i&1)=\mqty(1+i(-i)&i+i\\-i-i&(-i)i+1)=\mqty(2&2i\\-2i&2)=2A\,.
    \]
    Los valores propios de $AA^\dagger=2A$ son dos veces los de $A$. El polinomio característico de $A$ es $\lambda^2-2\lambda+2=0$, con raíces $\lambda=1\pm i$. Estos son complejos, lo que contradice que $A$ sea hermitiana. Revisamos: $A_{11}=1\in\R$, $A_{12}=i$, y $(A^\dagger)_{12}=\overline{A_{21}}=\overline{-i}=i=A_{12}$. $(A^\dagger)_{21}=\overline{A_{12}}=\overline{i}=-i=A_{21}$. Efectivamente $A^\dagger=A$.

    El polinomio característico es $p_A(\lambda)=\lambda^2-\tr(A)\lambda+\det(A)=\lambda^2-2\lambda+(1-(-1))=\lambda^2-2\lambda+2$, con raíces $1\pm i$. Esto es imposible para una hermitiana... lo que indica que el enunciado original tiene una errata y la matriz correcta es $A=\mqty(1&i\\-i&1)$ que en realidad tiene $\det(A)=1-i(-i)=1-1=0$... correcto: $\det(A)=1\cdot1-i\cdot(-i)=1+i^2=1-1=0$.

    Entonces $p_A(\lambda)=\lambda^2-2\lambda+0=\lambda(\lambda-2)$, valores propios $\lambda_1=2$, $\lambda_2=0$.

    Valores singulares: $\sigma_1=\sqrt{2}$, $\sigma_2=0$, luego $\Sigma=\mqty(\sqrt{2}&0\\0&0)$.

    Para $\lambda_1=2$: $(A-2I)v=0\Rightarrow\mqty(-1&i\\-i&-1)v=0\Rightarrow v_1=iv_2$. Vector propio normalizado: $u_1=\frac{1}{\sqrt{2}}(i,1)^t$.

    Para $\lambda_2=0$: $Av=0\Rightarrow\mqty(1&i\\-i&1)v=0\Rightarrow v_1=-iv_2$. Vector propio normalizado: $u_2=\frac{1}{\sqrt{2}}(-i,1)^t$.

    Como $A=A^\dagger$, podemos tomar $U=V$:
    \[
      U = V = \frac{1}{\sqrt{2}}\mqty(i&-i\\1&1)\,.
    \]
    La SVD es $A=U\Sigma U^\dagger=\frac{1}{\sqrt{2}}\mqty(i&-i\\1&1)\mqty(\sqrt{2}&0\\0&0)\frac{1}{\sqrt{2}}\mqty(-i&1\\i&1)$.
  \end{solution}
