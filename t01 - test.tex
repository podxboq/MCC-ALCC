  \question El conjugado de $z = 3 - 2i$ es:

  \begin{choices}
    \choice $-3 + 2i$
    \CorrectChoice $3 + 2i$
    \choice $-3 - 2i$
    \choice $2 + 3i$
  \end{choices}
  \begin{solution}
    El conjugado de $z = a + bi$ se obtiene cambiando el signo de la parte imaginaria, por tanto $\conj{3 - 2i} = 3 + 2i$.
  \end{solution}

  \question El módulo del número complejo $z = 3 + 4i$ es:

  \begin{choices}
    \choice $7$
    \choice $\sqrt{7}$
    \CorrectChoice $5$
    \choice $25$
  \end{choices}
  \begin{solution}
    El módulo se calcula como $|z| = \sqrt{3^2 + 4^2} = \sqrt{9 + 16} = \sqrt{25} = 5$.
  \end{solution}

  \question La fórmula de Euler establece que:

  \begin{choices}
    \choice $e^{i\theta} = \sin\theta + i\cos\theta$
    \CorrectChoice $e^{i\theta} = \cos\theta + i\sin\theta$
    \choice $e^{i\theta} = \cos\theta - i\sin\theta$
    \choice $e^{i\theta} = i\cos\theta + \sin\theta$
  \end{choices}
  \begin{solution}
    La fórmula de Euler es $e^{i\theta} = \cos\theta + i\sin\theta$.
  \end{solution}

  \question El producto $(1 + i)(2 - i)$ es igual a:

  \begin{choices}
    \choice $1 + i$
    \choice $2 - i$
    \CorrectChoice $3 + i$
    \choice $3 - i$
  \end{choices}
  \begin{solution}
    $(1 + i)(2 - i) = 2 - i + 2i - i^2 = 2 + i + 1 = 3 + i$.
  \end{solution}

  \question Las raíces cuartas de la unidad son:

  \begin{choices}
    \choice $1, -1, i, -i, e^{i\pi/4}$
    \CorrectChoice $1, i, -1, -i$
    \choice $1, i, -i, e^{i\pi/2}$
    \choice $1, -1, 2i, -2i$
  \end{choices}
  \begin{solution}
    Las raíces $n$-ésimas de la unidad son $e^{2\pi ik/n}$ para $k = 0, 1, \ldots, n-1$. Para $n = 4$: $1,\, i,\, -1,\, -i$.
  \end{solution}

  \question El número complejo $z = 1 + i$ en forma exponencial es:

  \begin{choices}
    \choice $\sqrt{2}e^{i\pi}$
    \CorrectChoice $\sqrt{2}e^{i\pi/4}$
    \choice $2e^{i\pi/4}$
    \choice $e^{i\pi/2}$
  \end{choices}
  \begin{solution}
    Módulo: $|z| = \sqrt{2}$. Argumento: $\Arg z = \arctan(1/1) = \pi/4$. Por tanto $z = \sqrt{2}e^{i\pi/4}$.
  \end{solution}

  \question Si $z = 2e^{i\pi/3}$, entonces $z^3$ es igual a:

  \begin{choices}
    \choice $6e^{i\pi}$
    \CorrectChoice $-8$
    \choice $8e^{i\pi/9}$
    \choice $2e^{i\pi}$
  \end{choices}
  \begin{solution}
    Por De Moivre: $z^3 = 2^3 e^{i3\pi/3} = 8e^{i\pi} = -8$.
  \end{solution}

  \question ¿Cuál es el valor de $\cos(i)$?

  \begin{choices}
    \choice $-1$
    \CorrectChoice $\dfrac{1+e^2}{2e}$
    \choice $i\cos(e)$
    \choice $1+\sin(e)$
  \end{choices}
  \begin{solution}
    $\cos(i) = \dfrac{e^{i^2} + e^{-i^2}}{2} = \dfrac{e^{-1} + e^{1}}{2} = \dfrac{1+e^2}{2e}$.
  \end{solution}

  \question Una amplitud cuántica $\alpha = \dfrac{3}{5} + \dfrac{4}{5}i$ tiene probabilidad asociada:

  \begin{choices}
    \choice $\dfrac{7}{5}$
    \CorrectChoice $1$
    \choice $\dfrac{9}{25}$
    \choice $\dfrac{16}{25}$
  \end{choices}
  \begin{solution}
    La probabilidad es $|\alpha|^2 = \left(\dfrac{3}{5}\right)^2 + \left(\dfrac{4}{5}\right)^2 = \dfrac{9}{25} + \dfrac{16}{25} = \dfrac{25}{25} = 1$.

    Obsérvese además que $|\alpha|=1$, por lo que $\alpha$ está sobre la circunferencia unidad.
  \end{solution}

  \question Para que dos amplitudes $\alpha_1 = \dfrac{1}{\sqrt{2}}e^{i\theta_1}$ y $\alpha_2 = \dfrac{1}{\sqrt{2}}e^{i\theta_2}$ interfieran de forma completamente destructiva, la diferencia de fases $\theta_2-\theta_1$ debe ser:

  \begin{choices}
    \choice $0$
    \choice $\dfrac{\pi}{2}$
    \CorrectChoice $\pi$
    \choice $\dfrac{\pi}{4}$
  \end{choices}
  \begin{solution}
    Para interferencia destructiva total necesitamos $\alpha_1 + \alpha_2 = 0$, es decir, $\alpha_2 = -\alpha_1$. Esto ocurre cuando $e^{i\theta_2} = -e^{i\theta_1} = e^{i(\theta_1+\pi)}$, es decir, $\theta_2 - \theta_1 = \pi$.

    Se puede verificar: $\alpha_1+\alpha_2 = \dfrac{1}{\sqrt{2}}e^{i\theta_1}(1+e^{i\pi}) = \dfrac{1}{\sqrt{2}}e^{i\theta_1}\cdot 0 = 0$.
  \end{solution}
