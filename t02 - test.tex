  \question El conjunto $W = \{(x,y,z) \mid x + y = 1\}$:

  \begin{choices}
    \choice Es un subespacio vectorial de $\C^3$.
    \CorrectChoice No es un subespacio vectorial de $\C^3$.
    \choice Es un espacio vectorial completo.
    \choice Es la base canónica de $\C^3$.
  \end{choices}
  \begin{solution}
    No es subespacio porque no contiene al vector cero: para $x=y=0$ se tiene $0+0=0\neq 1$.
  \end{solution}

  \question La dimensión de $\C^{2\times 2}$ (matrices complejas $2\times 2$) es:

  \begin{choices}
    \CorrectChoice $4$.
    \choice $2$.
    \choice $16$.
    \choice $8$.
  \end{choices}
  \begin{solution}
    Una base de $M_{2\times 2}(\C)$ son las cuatro matrices con un único $1$ en cada posición, luego la dimensión es $4$.
  \end{solution}

  \question Los vectores $(1,i)^t$ e $(i,1)^t$ en $\C^2$ son:

  \begin{choices}
    \choice Linealmente dependientes.
    \choice Ni independientes ni dependientes.
    \CorrectChoice Linealmente independientes.
    \choice Forman una base de $\C^3$.
  \end{choices}
  \begin{solution}
    De $\alpha(1,i)^t+\beta(i,1)^t=0$ se obtiene $\alpha+\beta i=0$ y $\alpha i+\beta=0$. De la primera, $\alpha=-\beta i$; sustituyendo en la segunda: $-\beta i^2+\beta=2\beta=0$, luego $\beta=\alpha=0$. Son independientes.
  \end{solution}

  \question El vector cero respecto de cualquier base tiene coordenadas:

  \begin{choices}
    \choice Distintas dependiendo de la base.
    \choice No se puede representar en ninguna base.
    \choice La primera coordenada es $1$ y el resto $0$.
    \CorrectChoice $(0,\ldots,0)$.
  \end{choices}
  \begin{solution}
    La representación del vector cero en cualquier base $\{v_1,\ldots,v_n\}$ es $0 = 0\cdot v_1+\cdots+0\cdot v_n$, luego todas las coordenadas son $0$.
  \end{solution}

  \question Si $U$ y $W$ son subespacios de un espacio vectorial $V$ con $\dim U = 2$ y $\dim W = 3$, entonces $\dim V$ es:

  \begin{choices}
    \choice Exactamente $5$.
    \choice Como mínimo $2$.
    \CorrectChoice Como mínimo $3$.
    \choice Como máximo $3$.
  \end{choices}
  \begin{solution}
    Como $W\subseteq V$ y $\dim W=3$, necesariamente $\dim V\geq 3$.  No se puede precisar más sin conocer la relación entre $U$ y $W$.
  \end{solution}

  \question Los vectores $\{v_1,v_2\}$ de $\C^2$ forman una base si y solo si:

  \begin{choices}
    \choice Son dos vectores cualesquiera de $\C^2$.
    \choice Son ortogonales.
    \CorrectChoice Son linealmente independientes.
    \choice Tienen módulo $1$.
  \end{choices}
  \begin{solution}
    En $\C^2$ (dimensión $2$), por la propiedad fundamental de la dimensión, dos vectores linealmente independientes siempre forman una base. La ortogonalidad y la norma unitaria son propiedades adicionales que no se exigen para ser base.
  \end{solution}

  \question El vector $\psi = \dfrac{1}{\sqrt{2}}\mqty(1\\1)$ expresado en la base computacional $\{e_0,e_1\}$ tiene coordenadas:

  \begin{choices}
    \choice $(1,1)$.
    \choice $(1,0)$.
    \CorrectChoice $\left(\dfrac{1}{\sqrt{2}},\,\dfrac{1}{\sqrt{2}}\right)$.
    \choice $(0,1)$.
  \end{choices}
  \begin{solution}
    Las coordenadas de $\psi$ en la base canónica $\{e_0,e_1\}$ coinciden directamente con sus componentes: $\psi = \dfrac{1}{\sqrt{2}}e_0+\dfrac{1}{\sqrt{2}}e_1$, luego las coordenadas son $\left(\dfrac{1}{\sqrt{2}},\dfrac{1}{\sqrt{2}}\right)$.
  \end{solution}

  \question Los subespacios $V_0=\gen\{e_0\}$ y $V_1=\gen\{e_1\}$ de $\C^2$ satisfacen:

  \begin{choices}
    \choice $V_0\cap V_1 = \C^2$.
    \choice $V_0 + V_1 \subsetneq \C^2$.
    \CorrectChoice $\C^2 = V_0\oplus V_1$.
    \choice $\dim(V_0\cap V_1) = 1$.
  \end{choices}
  \begin{solution}
    Como $V_0\cap V_1=\{0\}$ y todo vector $(\alpha,\beta)^t=\alpha e_0+\beta e_1\in V_0+V_1$, la suma es directa y es el espacio total: $\C^2=V_0\oplus V_1$.
  \end{solution}
