  \question
  Determinar si los siguientes conjuntos son subespacios de $\C^3$. Justificar la respuesta en cada caso.
  \begin{parts}
    \part $W_1 = \{(x,y,z)\in\C^3 \mid x + 2y - z = 1\}$.
    \part $W_2 = \{(x,y,z)\in\C^3 \mid x = \conj{z}\}$.
    \part $W_3 = \{(x,y,z)\in\C^3 \mid x + 2y - z = 0\}$.
    \part $W_4 = \{(x,y,z)\in\C^3 \mid |x|^2 + |y|^2 + |z|^2 = 1\}$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part No es subespacio: el vector cero no pertenece a $W_1$, pues $0+0-0 = 0\neq 1$.
      \part No es subespacio: si $v=(x,y,z)\in W_2$ entonces $x=\bar{z}$, pero $iv$ satisface $(iv)_1=ix$ e $(iv)_3=iz$, y $ix=\overline{iz} = -i\bar{z}=-ix$ implica $ix=-ix$, es decir, $x=0$; esto no se cumple para todo elemento de $W_2$ (por ejemplo $(1,0,1)\in W_2$ pero $i(1,0,1)=(i,0,i)\notin W_2$ pues $i\neq -i$).
      \part Sí es subespacio. Sean $u=(x_1,y_1,z_1),v=(x_2,y_2,z_2)\in W_3$ y $\alpha,\beta\in\C$. Entonces
      \[
        (\alpha x_1+\beta x_2)+2(\alpha y_1+\beta y_2)-(\alpha z_1+\beta z_2)
        = \alpha(x_1+2y_1-z_1)+\beta(x_2+2y_2-z_2) = 0\,.
      \]
      \part No es subespacio: $2v\in W_4$ requeriría $|2x|^2+|2y|^2+|2z|^2=4\neq 1$.
    \end{parts}
  \end{solution}

  \question
  Encontrar una base y la dimensión del subespacio $W$ de $\C^4$ generado por los vectores
  \[
    v_1=\mqty(1\\i\\0\\1)\,,\quad v_2=\mqty(i\\1\\1\\0)\,,\quad v_3=\mqty(1+i\\1+i\\1\\1)\,.
  \]
  \begin{solution}
    Observamos que $v_3 = v_1+v_2$: en efecto, componente a componente, $1+i = 1+i$, $1+i=i+1$, $1=0+1$, $1=1+0$. Por tanto, los tres vectores son linealmente dependientes.

    Sin embargo, $v_1$ y $v_2$ son linealmente independientes: de $\alpha v_1+\beta v_2=0$ obtenemos el sistema $\alpha+\beta i=0$, $\alpha i+\beta=0$. De la primera ecuación $\alpha=-\beta i$; sustituyendo en la segunda: $-\beta i^2+\beta=2\beta=0$, luego $\beta=\alpha=0$.

    Por tanto, $\mathcal{B}=\{v_1,v_2\}$ es una base de $W$ y $\dim W = 2$.
  \end{solution}

  \question
  Verificar que $\{v_1,v_2,v_3\}$ es linealmente independiente en $\C^3$ y extenderlo a una base de $\C^3$ para
  \[
    v_1=\mqty(1\\1\\0)\,,\quad v_2=\mqty(1\\i\\0)\,,\quad v_3=\mqty(0\\0\\1)\,.
  \]
  \begin{solution}
    De $\alpha v_1+\beta v_2+\gamma v_3=0$:
    \begin{align*}
      \alpha+\beta   & = 0\,, \\
      \alpha+\beta i & = 0\,, \\
      \gamma         & = 0\,.
    \end{align*}
    De la tercera, $\gamma=0$. Restando la primera de la segunda: $\beta(i-1)=0$, luego $\beta=0$ y por tanto $\alpha=0$. Los tres vectores son linealmente independientes.

    Como $\dim\C^3=3$ y tenemos tres vectores linealmente independientes, por las propiedades de la dimensión, $\{v_1,v_2,v_3\}$ ya es una base de $\C^3$ y no es necesario añadir más vectores.
  \end{solution}

  \question
  Demostrar que si $U$ y $W$ son subespacios de $\C^3$ con $\dim U+\dim W = 4$, entonces $U\cap W \neq \{0\}$.
  \begin{solution}
    Por la fórmula de la dimensión~\ref{prop:formula-dimension}:
    \[
      \dim(U\cap W) = \dim U + \dim W - \dim(U+W)\,.
    \]
    Como $U+W\subseteq\C^3$, se tiene $\dim(U+W)\leq 3$. Por tanto,
    \[
      \dim(U\cap W) = 4 - \dim(U+W) \geq 4-3 = 1 > 0\,.
    \]
    Un subespacio de dimensión positiva contiene vectores no nulos, luego $U\cap W\neq\{0\}$.
  \end{solution}

  \question \label{ejer:propiedades-lineal-independencia}
  Verificar las propiedades de la independencia lineal enunciadas en la proposición~\ref{prop:propiedades-lineal-independencia}. Para cada propiedad, construir un ejemplo en $\C^2$ o $\C^3$ que la ilustre.
  \begin{solution}
    \begin{enumerate}
      \item Si $0\in\{v_1,\ldots,v_k\}$, digamos $v_1=0$, la relación $1\cdot v_1+0\cdots=0$ es no trivial. Ejemplo: $\{(0,0)^t,(1,i)^t\}$ es dependiente.
      \item Si $\{v_1,v_2,v_3\}$ es linealmente independiente, entonces $\{v_1,v_2\}$ también lo es (si hubiera relación $\alpha v_1+\beta v_2=0$ no trivial, añadiendo $0\cdot v_3$ obtendríamos una relación no trivial en el conjunto mayor).
      \item En $\C^2$ ($n=2$), los tres vectores $(1,0)^t,(0,1)^t,(1,1)^t$ son dependientes: $(1,1)^t=(1,0)^t+(0,1)^t$.
      \item En $\C^2$, $(1,0)^t$ y $(0,1)^t$ generan todo $\C^2$: forman un conjunto generador mínimo (dos vectores = dimensión).
    \end{enumerate}
  \end{solution}

  \question
  Sean $\mathcal{B}$ y $\mathcal{B}'$ bases de $\C^2$ formadas por
  \[
    \mathcal{B} = \left\{\mqty(1\\i),\mqty(i\\1)\right\}\,,\qquad
    \mathcal{B}' = \left\{\mqty(i\\0),\mqty(0\\i)\right\}\,.
  \]
  \begin{parts}
    \part Hallar la matriz de cambio de coordenadas de $\mathcal{B}$ a $\mathcal{B}'$.
    \part Hallar la matriz de cambio de coordenadas de $\mathcal{B}'$ a $\mathcal{B}$.
    \part Verificar que ambas matrices son inversas una de la otra.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part Las matrices de cambio a la base canónica $\mathcal{C}$ son
      \[
        M_{\mathcal{B}}^{\mathcal{C}} = \mqty(1 & i \\ i & 1)\,,\qquad M_{\mathcal{B}'}^{\mathcal{C}} = \mqty(i & 0 \\ 0 & i)\,.
      \]
      Calculamos $\bigl(M_{\mathcal{B}'}^{\mathcal{C}}\bigr)^{-1} = \frac{1}{i^2}\mqty(i & 0 \\ 0 & i) = -\frac{1}{1}\mqty(i & 0 \\ 0 & i)^{-1}$. Como $M_{\mathcal{B}'}^{\mathcal{C}} = i\,I$, su inversa es $\frac{1}{i}I = -iI$. Por tanto:
      \[
        M_{\mathcal{B}}^{\mathcal{B}'} = M_{\mathcal{B}}^{\mathcal{C}}\cdot(-iI) = -i\mqty(1 & i \\ i & 1) = \mqty(-i & 1 \\ 1 & -i)\,.
      \]
      \part $M_{\mathcal{B}'}^{\mathcal{B}} = \bigl(M_{\mathcal{B}}^{\mathcal{B}'}\bigr)^{-1}$. El determinante de $M_{\mathcal{B}}^{\mathcal{B}'}$ es $(-i)(-i)-1\cdot1=-1-1=-2$. Por tanto:
      \[
        M_{\mathcal{B}'}^{\mathcal{B}} = \frac{1}{-2}\mqty(-i & -1 \\ -1 & -i) = \frac{1}{2}\mqty(i & 1 \\ 1 & i)\,.
      \]
      \part Comprobación:
      \[
        M_{\mathcal{B}}^{\mathcal{B}'}\cdot M_{\mathcal{B}'}^{\mathcal{B}}
        = \mqty(-i & 1 \\ 1 & -i)\cdot\frac{1}{2}\mqty(i & 1 \\ 1 & i)
        = \frac{1}{2}\mqty(-i^2+1 & -i+i \\ i-i & 1-i^2)
        = \frac{1}{2}\mqty(2 & 0 \\ 0 & 2) = I\,.\quad\checkmark
      \]
    \end{parts}
  \end{solution}

  \question
  Sean $U = \gen{(1,0,1)^t,(0,1,0)^t}$ y $W = \gen{(1,1,1)^t,(0,1,1)^t}$ subespacios de $\C^3$.
  \begin{parts}
    \part Calcular $\dim U$, $\dim W$, $\dim(U+W)$ y $\dim(U\cap W)$.
    \part Encontrar una base de $U\cap W$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part Los generadores de $U$ son claramente independientes (ninguno es múltiplo del otro y tienen ceros en posiciones distintas), luego $\dim U = 2$. Análogamente, los generadores de $W$ son independientes (si $\alpha(1,1,1)^t+\beta(0,1,1)^t=0$ entonces $\alpha=0$ de la primera componente, y $\beta=0$ de la segunda), luego $\dim W=2$.

      Para $U+W$: los cuatro vectores generadores de $U$ y $W$ son $(1,0,1)^t,(0,1,0)^t,(1,1,1)^t,(0,1,1)^t$. Observamos que $(1,1,1)^t=(1,0,1)^t+(0,1,0)^t\in U$, de modo que $W\subseteq U$. Entonces $U+W=U$ y $\dim(U+W)=2$.

      Por la fórmula: $\dim(U\cap W)=2+2-2=2$, luego $U\cap W=W=U$.
      \part Como $U=W$, una base de $U\cap W$ es $\{(1,0,1)^t,(0,1,0)^t\}$.
    \end{parts}
  \end{solution}

  \question
  Sea $\{e_0, e_1\}$ la base computacional de $\C^2$.
  \begin{parts}
    \part Expresar los vectores $\psi_1 = \dfrac{1}{\sqrt{2}}\mqty(1\\1)$, $\psi_2 = \dfrac{1}{\sqrt{2}}\mqty(1\\-1)$ y $\psi_3 = \dfrac{1}{\sqrt{2}}\mqty(1\\i)$ como combinaciones lineales de $e_0$ y $e_1$, identificando las amplitudes $\alpha$ y $\beta$ en cada caso.
    \part Verificar que $\{\psi_1, \psi_2\}$ es una base de $\C^2$.
    \part Expresar $e_0$ y $e_1$ como combinaciones lineales de $\psi_1$ y $\psi_2$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part Las coordenadas en la base computacional son directamente las componentes del vector:
      \[
        \psi_1 = \frac{1}{\sqrt{2}}e_0 + \frac{1}{\sqrt{2}}e_1\,,\quad
        \psi_2 = \frac{1}{\sqrt{2}}e_0 - \frac{1}{\sqrt{2}}e_1\,,\quad
        \psi_3 = \frac{1}{\sqrt{2}}e_0 + \frac{i}{\sqrt{2}}e_1\,.
      \]

      \part Comprobamos la independencia lineal de $\{\psi_1,\psi_2\}$. De $\alpha\psi_1+\beta\psi_2=0$:
      \[
        \frac{\alpha+\beta}{\sqrt{2}}=0\quad\text{y}\quad\frac{\alpha-\beta}{\sqrt{2}}=0\,,
      \]
      de donde $\alpha=\beta=0$. Son independientes. Como $\dim\C^2=2$, forman una base.

      \part Sumando y restando: $\psi_1+\psi_2 = \sqrt{2}\,e_0$ y $\psi_1-\psi_2=\sqrt{2}\,e_1$, luego
      \[
        e_0 = \frac{1}{\sqrt{2}}\psi_1+\frac{1}{\sqrt{2}}\psi_2\,,\qquad
        e_1 = \frac{1}{\sqrt{2}}\psi_1-\frac{1}{\sqrt{2}}\psi_2\,.
      \]
    \end{parts}
  \end{solution}

  \question
  Se definen los subespacios $V_0 = \gen{e_0 + e_1}$ y $V_1 = \gen{e_0 -e_1}$ de $\C^2$.
  \begin{parts}
    \part Demostrar que $\C^2 = V_0\oplus V_1$.
    \part Para el vector $\psi = \mqty(\alpha\\\beta)$, encontrar su descomposición única $\psi = v_0 + v_1$ con $v_0\in V_0$ y $v_1\in V_1$.
    \part Calcula la amplitud y fase de $e_0$ en la base $\mathcal{B}=\{e_0+e_1,e_0-e_1\}$.
  \end{parts}
  \begin{solution}
    \begin{parts}
      \part Comprobamos las dos condiciones de suma directa:

      \textit{Suma:} Todo vector $\psi=(\alpha,\beta)^t = \frac{\alpha+\beta}{2}(e_0+e_1)+\frac{\alpha-\beta}{2}(e_0-e_1) \in V_0+V_1$.

      \textit{Intersección nula:} Si $v\in V_0\cap V_1$, entonces $v=\alpha(e_0+e_1)=\beta(e_0-e_1)$ para ciertos $\alpha,\beta\in\C$. Igualando componentes: $\alpha+\beta=\beta-\alpha$ y $\alpha+\beta=-\beta+\alpha$, luego $v=0$. Por tanto $V_0\cap V_1=\{0\}$ y la suma es directa.

      \part $v_0 = \frac{\alpha+\beta}{2}(e_0+e_1)$ y $v_1 = \frac{\alpha-\beta}{2}(e_0-e_1)$. La unicidad se garantiza por ser suma directa (tema~\ref{prop:formula-dimension}).

      \part En la base $\mathcal{B}=\{e_0+e_1,e_0-e_1\}$, $e_0 = \frac{1}{2}(e_0+e_1)+\frac{1}{2}(e_0-e_1)$. Por tanto, la amplitud de $e_0$ es $\frac{1}{2}$ y la fase es $0$.
    \end{parts}
  \end{solution}
